__author__ = "Gabriel Hishida and Allan Cedric"

import random
import socket
import os

HOST = "127.0.0.1" # The server's hostname or IP address
PORTS = [64000, 64001, 64002] # Each port is used by one server

# Generate a random temperature
def get_temperature(weather):
    if weather == "hot":
        return random.randint(40, 55)
    else:
        return random.randint(-50, -20)

if __name__ == "__main__":
    retval = os.fork() # Get First + Second

    if retval != 0:
        retval = os.fork() # Get Third

        if retval != 0: # First
            PORT = PORTS[0]
            weather = "hot"
        else: # Third
            PORT = PORTS[2]
            weather = "cold"

    else: # Second
        PORT = PORTS[1]
        weather = "hot"

    # if random.randint(0, 1):
    #     weather = "hot"
    # else:
    #     weather = "cold"

    # Create the server's socket
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        
        # Internet, TCP
        s.bind((HOST, PORT))
        
        # Wait for client
        s.listen()
        conn, addr = s.accept()
        with conn:
            print(f">>> Server {PORT} connected to CACHE on socket {addr}")
            while True:
                data = conn.recv(1024)
                data = data.decode("utf-8")
                if not data:
                    print("No connection. Finishing...")
                    break

                print(f"Received request `{data}`")
                response = str(get_temperature(weather))
                print(f"Sending response `{response}`")
                conn.sendall(response.encode("utf-8"))
